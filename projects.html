Blazor Task Manager

Course: CIS 293

Description

This project is a task management web application built with Blazor. It allows users to create, edit, and track their daily tasks with features like due dates, priority levels, and completion status. The app uses Entity Framework Core for data access and follows the MVC architectural pattern.

Screenshots
Sample Code
using System.ComponentModel.DataAnnotations;

namespace TaskManagerApp.Models
{
    public class TaskItem
    {
        public int Id { get; set; }

        [Required(ErrorMessage = "Title is required")]
        [StringLength(100, ErrorMessage = "Title can't be longer than 100 characters")]
        public string Title { get; set; } = string.Empty;

        public string? Description { get; set; }

        [Required(ErrorMessage = "Due date is required")]
        public DateTime DueDate { get; set; } = DateTime.Today.AddDays(1);

        public PriorityLevel Priority { get; set; } = PriorityLevel.Medium;

        public bool IsCompleted { get; set; } = false;

        public DateTime CreatedAt { get; set; } = DateTime.Now;
    }

    public enum PriorityLevel
    {
        Low,
        Medium,
        High
    }
}
      
using Microsoft.EntityFrameworkCore;
using TaskManagerApp.Data;
using TaskManagerApp.Models;

namespace TaskManagerApp.Data
{
    public class TaskService
    {
        private readonly AppDbContext _context;

        public TaskService(AppDbContext context)
        {
            _context = context;
        }

        public async Task> GetAllTasksAsync()
        {
            return await _context.Tasks.OrderBy(t => t.DueDate).ToListAsync();
        }

        public async Task GetTaskByIdAsync(int id)
        {
            return await _context.Tasks.FindAsync(id);
        }

        public async Task AddTaskAsync(TaskItem task)
        {
            _context.Tasks.Add(task);
            await _context.SaveChangesAsync();
        }

        public async Task UpdateTaskAsync(TaskItem task)
        {
            _context.Tasks.Update(task);
            await _context.SaveChangesAsync();
        }

        public async Task DeleteTaskAsync(int id)
        {
            var task = await _context.Tasks.FindAsync(id);
            if (task != null)
            {
                _context.Tasks.Remove(task);
                await _context.SaveChangesAsync();
            }
        }
        public async Task ToggleCompleteAsync(int id)
        {
            var task = await _context.Tasks.FindAsync(id);
            if (task != null)
            {
                task.IsCompleted = !task.IsCompleted;
                await _context.SaveChangesAsync();
            }
        }
    }
}

          class Employee:
    def __init__(self, name, rate):
        self.name = name
        self.hourly_rate = rate
        self.hours_worked = 0

    def calculate_pay(self):
        return self.hourly_rate * self.hours_worked
        
    def add_employee(self):
        name = simpledialog.askstring("Add", "Employee Name:")
        if not name:
            return
        rate = simpledialog.askfloat("Add", f"Hourly Rate for {name}:")
        if rate is None:
            return
        Company.employees.append(Employee(name, rate))
        self.update_listbox()
        messagebox.showinfo("Added", f"{name} added!")

    def update_employee(self):
        sel = self.listbox.curselection()
        if not sel:
            messagebox.showwarning("Select", "Select an employee first")
            return
        emp = Company.employees[sel[0]]
        rate = simpledialog.askfloat("Update", f"New rate for {emp.name}:")
        if rate is None:
            return
        emp.hourly_rate = rate
        self.update_listbox()
        messagebox.showinfo("Updated", f"{emp.name}'s rate updated.")

Trivia Game

Course: CIS 293

Description

This project is a trivia game web application built with ASP.NET Core MVC and Entity Framework Core. Players take turns on a single computer answering randomized multiple choice questions under a countdown timer, earning points for each correct answer. After all players finish their questions a leaderboard displays the final rankings by score with average answer time used as a tiebreaker.

[HttpPost]
[ValidateAntiForgeryToken]
public async Task AddQuestion(int quizId, string questionText,
    string answerA, string answerB, string answerC, string answerD,
    string correctAnswer, int points, int timeLimitSeconds)
{
    var quiz = await _context.Quizzes.FirstOrDefaultAsync(q => q.Id == quizId);

    if (quiz == null)
    {
        return NotFound();
    }

    if (string.IsNullOrWhiteSpace(questionText) ||
        string.IsNullOrWhiteSpace(answerA) ||
        string.IsNullOrWhiteSpace(answerB) ||
        string.IsNullOrWhiteSpace(answerC) ||
        string.IsNullOrWhiteSpace(answerD) ||
        string.IsNullOrWhiteSpace(correctAnswer))
    {
        TempData["Error"] = "All fields are required.";
        return RedirectToAction(nameof(Edit), new { id = quizId });
    }

    var question = new Question
    {
        QuizId = quizId,
        QuestionText = questionText.Trim(),
        AnswerA = answerA.Trim(),
        AnswerB = answerB.Trim(),
        AnswerC = answerC.Trim(),
        AnswerD = answerD.Trim(),
        CorrectAnswer = correctAnswer.Trim().ToUpper(),
        Points = points,
        TimeLimitSeconds = timeLimitSeconds
    };

    _context.Questions.Add(question);
    await _context.SaveChangesAsync();

    TempData["Success"] = "Question added.";
    return RedirectToAction(nameof(Edit), new { id = quizId });
}
        
          const timeLimit = @Model.TimeLimitSeconds;
const startTime = Date.now();
let timerInterval;

const timerDisplay = document.getElementById('timerDisplay');
const timerBar = document.getElementById('timerBar');
const timeTakenInput = document.getElementById('timeTakenSeconds');
const form = document.getElementById('answerForm');

timerInterval = setInterval(() => {
    const elapsed = (Date.now() - startTime) / 1000;
    const remaining = Math.max(timeLimit - elapsed, 0);
    const percent = (remaining / timeLimit) * 100;

    timerDisplay.textContent = Math.ceil(remaining);
    timerBar.style.width = percent + '%';

    if (remaining <= 0) {
        clearInterval(timerInterval);
        timeTakenInput.value = timeLimit;
        form.submit();
    }
}, 250);

form.addEventListener('submit', function () {
    clearInterval(timerInterval);
    const elapsed = (Date.now() - startTime) / 1000;
    timeTakenInput.value = Math.min(Math.round(elapsed * 10) / 10, timeLimit);
});
