C# / ASP.NET

Case Study

Description

SportsPro is a web application built for a sports software company to help them manage their customer support operations. It lets admins keep track of customers, products, and technicians, while also allowing customers to register the products they own. When something goes wrong, support incidents can be opened and assigned to technicians to resolve.

Screenshots
Sample Code
using SportsPro.Models;

namespace SportsPro.DataLayer
{
    public interface IUnitOfWork
    {
        Repository Products { get; }
        Repository Technicians { get; }
        Repository Customers { get; }
        Repository Incidents { get; }
        Repository Countries { get; }
        Repository Registrations { get; }

        void Save();
    }
}     
using Microsoft.AspNetCore.Mvc;
using SportsPro.DataLayer;
using SportsPro.Models;

namespace SportsPro.Controllers
{
    public class TechIncidentController : Controller
    {
        private readonly IUnitOfWork _data;

        public TechIncidentController(IUnitOfWork data)
        {
            _data = data;
        }

        [HttpGet]
        public IActionResult Get()
        {
            var techOptions = new QueryOptions
            {
                OrderBy = t => t.Name
            };

            ViewBag.Technicians = _data.Technicians.List(techOptions);
            return View();
        }

        [HttpPost]
        public IActionResult Get(int technicianId)
        {
            if (technicianId == 0)
            {
                var techOptions = new QueryOptions
                {
                    OrderBy = t => t.Name
                };

                ViewBag.Technicians = _data.Technicians.List(techOptions);
                ViewBag.ErrorMessage = "Please select a technician.";
                return View();
            }

            HttpContext.Session.SetInt32("TechnicianId", technicianId);
            return RedirectToAction("List");
        }

        [HttpGet]
        public IActionResult List()
        {
            int? techId = HttpContext.Session.GetInt32("TechnicianId");

            if (techId == null)
            {
                return RedirectToAction("Get");
            }

            var technician = _data.Technicians.Get(techId.Value);

            var incidentOptions = new QueryOptions
            {
                Includes = "Customer,Product",
                Where = i => i.TechnicianID == techId.Value,
                Where2 = i => i.DateClosed == null
            };

            var incidents = _data.Incidents.List(incidentOptions);

            ViewBag.Technician = technician;
            return View(incidents);
        }

        [HttpGet]
        public IActionResult Edit(int id)
        {
            var options = new QueryOptions
            {
                Includes = "Technician,Customer,Product",
                Where = i => i.IncidentID == id
            };

            var incident = _data.Incidents.Get(options);
            return View(incident);
        }

        [HttpPost]
        public IActionResult Edit(Incident incident)
        {
            if (ModelState.IsValid)
            {
                _data.Incidents.Update(incident);
                _data.Save();
                return RedirectToAction("List");
            }

            var options = new QueryOptions
            {
                Includes = "Technician,Customer,Product",
                Where = i => i.IncidentID == incident.IncidentID
            };

            var currentIncident = _data.Incidents.Get(options);
            return View(currentIncident);
        }

        public IActionResult Switch()
        {
            HttpContext.Session.Remove("TechnicianId");
            return RedirectToAction("Get");
        }
    }
}
C# / .NET

Quarterly Sales Application

Description

This web application displays quarterly sales data on the home page with a dropdown to filter results by employee, and includes separate pages for adding new employees and sales records. Employee entries require valid first name, last name, date of birth, date of hire, and manager, preventing duplicate employee records. Sales entries require a valid quarter between 1 and 4, a year after 2000, a positive amount, and must not duplicate an existing entries.

Screenshots
Sample Code
using System.ComponentModel.DataAnnotations;

namespace QuarterlySales.Models
{
    public class Sales
    {
        public int SalesId { get; set; }

        [Required]
        public int Quarter { get; set; }

        [Required]
        public int Year { get; set; }

        [Required]
        public decimal Amount { get; set; }

        [Required]
        public int EmployeeId { get; set; }

        public Employee? Employee { get; set; }
    }
}
        
using Microsoft.AspNetCore.Mvc;
using Microsoft.AspNetCore.Mvc.Rendering;
using QuarterlySales.Data;
using QuarterlySales.Models;
using QuarterlySales.ViewModels;

namespace QuarterlySales.Controllers
{
    public class SalesController : Controller
    {
        private readonly QuarterlySalesContext _context;

        public SalesController(QuarterlySalesContext context)
        {
            _context = context;
        }

        [HttpGet]
        public IActionResult Add()
        {
            LoadEmployees();
            return View(new SalesViewModel());
        }

        [HttpPost]
        public IActionResult Add(SalesViewModel vm)
        {
            if (!ModelState.IsValid)
            {
                LoadEmployees();
                return View(vm);
            }

            var sales = new Sales
            {
                Quarter    = vm.Quarter,
                Year       = vm.Year,
                Amount     = vm.Amount,
                EmployeeId = vm.EmployeeId
            };

            _context.Sales.Add(sales);
            _context.SaveChanges();

            return RedirectToAction("Index", "Home");
        }

        private void LoadEmployees()
        {
            var employees = _context.Employees.OrderBy(e => e.Lastname).ToList();
            ViewBag.Employees = new SelectList(employees, "EmployeeId", "FullName");
        }
    }
}
C# / .NET

Price Quotation

Description

Using Bootstrap for styling, this web application allows users to select a product, enter a quantity, and receive a price quote.

Screenshots
Sample Code
using System.ComponentModel.DataAnnotations;

namespace PriceQuotation.Models
{
    public class Quotation
    {
        [Required(ErrorMessage = "Please enter a sale price.")]
        [Range(0, 100000000, ErrorMessage = "Sale price must be greater than zero.")]
        public double? Subtotal { get; set; }

        [Required(ErrorMessage = "Please enter a discount percent.")]
        [Range(0, 100, ErrorMessage = "Discount percent must be between 0 and 100.")]
        public double? DiscountPercent { get; set; }

        public double CalculateDiscount()
        {
            if (Subtotal.HasValue && DiscountPercent.HasValue)
            {
                return Subtotal.Value * (DiscountPercent.Value / 100);
            }
            else
            {
                return 0;
            }
        }

        public double CalculateTotal()
        {
            if (Subtotal.HasValue)
            {
                double discount = CalculateDiscount();
                return Subtotal.Value - discount;
            }
            else
            {
                return 0;
            }
        }
    }
}
        
          using Microsoft.AspNetCore.Mvc;
using PriceQuotation.Models;

namespace PriceQuotation.Controllers
{
    public class HomeController : Controller
    {
        [HttpGet]
        public IActionResult Index()
        {
            ViewBag.Discount = 0;
            ViewBag.Total = 0;
            return View();
        }

        [HttpPost]
        public IActionResult Index(Quotation quote)
        {
            if (ModelState.IsValid)
            {
                ViewBag.Discount = quote.CalculateDiscount();
                ViewBag.Total = quote.CalculateTotal();
            }
            else
            {
                ViewBag.Discount = 0;
                ViewBag.Total = 0;
            }
            return View(quote);
        }
    }
}
